typedef struct nod_arb 
{ int nr;				
   struct nod_arb *stg, *dr;
} NOD_ARB;
NOD_ARB *rad;
void preordine( NOD_ARB *p)
{  if (p) 
    { printf("%d ",p->nr);
       preordine(p->stg);
       preordine(p->dr);
}

void inordine( NOD_ARB *p)
{  if (p) 
    { inordine(p->stg);
       printf("%d ",p->nr);
       inordine(p->dr);
}

void postordine( NOD_ARB *p)
{  if (p) 
    { postordine(p->stg);
       postordine(p->dr);
       printf("%d ",p->nr);
}
